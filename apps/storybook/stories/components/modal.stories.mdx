# Modal

A component to display a `Modal`.

[View source](https://github.com/actarian/design-system/tree/main/src/ui/components/modal)

## Imports

```js
import { Modal } from '@websolute/ui';
```

## Example

A basic example of `Modal`.

export const Basic = ({ data, ...args }) => {
  const [modal, onOpenModal, onCloseModal] = useModal();
  return (<>
    <div style={{ margin: '0 1em' }}>
      <Button variant="default" marginBottom="1rem" onClick={() => onOpenModal('modal')}>Open Modal</Button>
      <Modal width="30rem" visible={modal == 'modal'} onClose={onCloseModal}>
        <Modal.Title>
          <Text size="8" fontWeight="700">Modal Title</Text>
        </Modal.Title>
        <Modal.Subtitle>Modal Subtitle</Modal.Subtitle>
        <Modal.Content>
          <p>Modal content lorem ispum.</p>
        </Modal.Content>
        <Modal.Button passive onClick={onCloseModal}>Cancel</Modal.Button>
        <Modal.Button>Confirm</Modal.Button>
      </Modal>
    </div>
  </>);
}
